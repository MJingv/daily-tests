const { should, expect, asserts } = require('chai')
 
describe('#math', () => {

    it('should return 5 when 2+3', () => {
      expect(add(2, 3), 5)
    })
    it('should return 5 when 2+3', () => {
      expect(add(-1, 1), 0)
    })
 
  })

    it('should return 6 when 2*3', () => {
      expect(add(2, 3), 6)
    })
    it('should return 0 when 0*2', () => {
      expect(add(0, 2), 0)
    })
 
  })
